```tsx
'use client';
import { useParams, useSearchParams } from 'next/navigation';
import Link from 'next/link';

export default function RepairDetailPage() {
  const params = useParams();
  const sp = useSearchParams();
  const id = String(params.id ?? '');

  // T-09-01: dev-only param gating
  const isDev = process.env.NODE_ENV !== 'production';
  const forceLoading = isDev && sp.get('loading') === '1';
  const forceError   = isDev && sp.get('error') === '1';

  // Find the entity by id-style key (per-page: reference / id / customerCode etc.)
  const job = repairs.find(r => r.reference === id);

  const ROUTE = '/repair/[id]';

  if (forceLoading) {
    return <AppShell><DetailSpinner label={getLoadingLabel(ROUTE)} /></AppShell>;
  }
  if (forceError) {
    const errCopy = getErrorCopy('/repair');  // detail pages fall back to the list-route errorCopy
    return (
      <AppShell>
        <ErrorState title={errCopy.title} description={errCopy.description}
                    onRetry={() => window.location.reload()} />
      </AppShell>
    );
  }
  if (!job) {
    // NotFound branch with Levenshtein suggestion (D-03)
    const allRefs = repairs.map(r => r.reference);
    const suggestion = nearestReference(id, allRefs);
    const copy = getEmptyCopy(ROUTE, 'not-found');
    if (!copy) return <AppShell><EmptyState variant="not-found" title="Not found" /></AppShell>;
    return (
      <AppShell>
        <EmptyState
          variant="not-found"
          icon={copy.icon}
          title={copy.title}
          description={
            <>
              {copy.description.replace('{ID}', id)}
              {suggestion && (
                <>
                  <br /><br />
                  Did you mean{' '}
                  <Link href={`/repair/${suggestion}`} className="gecko-text-mono"
                        style={{ color: 'var(--gecko-primary-600)' }}>
                    {suggestion}
                  </Link>
                  ?
                </>
              )}
            </>
          }
          primary={copy.primary}
          secondary={copy.secondary && {
            ...copy.secondary,
            href: copy.secondary.href.replace('{ID}', encodeURIComponent(id)),
          }}
        />
      </AppShell>
    );
  }

  // Existing data render — entity detail JSX
  return <AppShell>{/* existing detail JSX */}</AppShell>;
}
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| URL [id] param → seed-data lookup | The `id` is user-controllable; the lookup is `.find(r => r.reference === id)` (string-equality, no SQL/template-injection risk). |
| URL [id] param → NotFound description rendering | `{ID}` substitution into description string. Rendered via JSX children (not innerHTML). |
| URL query params → state branch | Same as plan 08 — dev-only gate via NODE_ENV. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-09-01 | Spoofing (phishing-via-URL) | Dev-param gate | mitigate | **HIGH** — same as T-08-01. Every `sp.get('loading')` / `sp.get('error')` MUST be gated by `process.env.NODE_ENV !== 'production'`. Acceptance criterion greps. |
| T-09-02 | Tampering (XSS via [id]) | NotFound description {ID} substitution | mitigate | `id` is rendered as React children (text), not innerHTML. The `secondary.href` substitution uses `encodeURIComponent(id)` so URL-special chars are escaped. Acceptance criterion: grep for `encodeURIComponent` near the `replace('{ID}'...)` patterns. |
| T-09-03 | Open redirect | nearestReference suggestion Link | accept | Suggestion is computed from in-memory seed array; href is `/repair/${suggestion}` where suggestion is from a hardcoded list. No external URLs. |
