```tsx
'use client';
import { useSearchParams } from 'next/navigation';

export default function EquipmentListPage() {
  const sp = useSearchParams();

  // Dev-mode-only param hooks per T-08-01 mitigation
  const isDev = process.env.NODE_ENV !== 'production';
  const forceLoading     = isDev && sp.get('loading') === '1';
  const forceError       = isDev && sp.get('error') === '1';
  const forceEmpty       = isDev && sp.get('empty') === '1';
  const forceFilterEmpty = isDev && sp.get('filter-empty') === '1';

  // Existing filter / search state — keep whatever the current page uses
  const typeFilter = sp.get('type');  // e.g. 'REEFER'
  const hasActiveFilters = !!typeFilter || /* other filters */;

  // Apply filters to seed data
  const filteredData = equipment.filter(/* ... */);
  const showFilterEmpty = forceFilterEmpty || (filteredData.length === 0 && hasActiveFilters);
  const showEmpty       = forceEmpty       || (filteredData.length === 0 && !hasActiveFilters);

  if (forceLoading) {
    return <AppShell><TableSkeleton columns={6} rows={8} /></AppShell>;
  }
  if (forceError) {
    const errCopy = getErrorCopy('/equipment');
    return (
      <AppShell>
        <ErrorState title={errCopy.title} description={errCopy.description}
                    onRetry={() => window.location.reload()} />
      </AppShell>
    );
  }
  if (showEmpty || showFilterEmpty) {
    const variant: EmptyStateVariant = showFilterEmpty ? 'filter-empty' : 'empty';
    const copy = getEmptyCopy('/equipment', variant);
    if (!copy) {
      // Fallback: copy missing → render generic message (shouldn't happen if plan 04 covered all routes)
      return <AppShell><EmptyState variant={variant} title="No data" /></AppShell>;
    }
    return (
      <AppShell>
        <EmptyState
          variant={variant}
          icon={copy.icon}
          title={copy.title}
          description={copy.description}
          primary={copy.primary}
          secondary={copy.secondary}
        />
      </AppShell>
    );
  }

  // Existing data render — table / kanban / etc.
  return <AppShell>{/* existing JSX */}</AppShell>;
}
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| URL query params → component render branch | `?loading=1` etc. are user-controllable strings from the URL. A malicious link could force any state. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-08-01 | Spoofing (phishing-via-URL) | Dev-mode query param gate | mitigate | **HIGH severity if mis-implemented.** Every dev-param check is gated by `process.env.NODE_ENV !== 'production'`. Production builds (`NEXT_PUBLIC_*` is per request — but `process.env.NODE_ENV` is fixed at build time) IGNORE the query params.
